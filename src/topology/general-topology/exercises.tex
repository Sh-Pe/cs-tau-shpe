	\section[Metric spaces]{}
	\nsubsection{Napkin 6.1.3}
	A metric space $M$ is bounded iff (definition) $\exists D \in \R .\, \forall p, q \in M \co d(p, q) \le D$ iff $\forall p \in M.\, \exists R_p \in \R.\,\forall q \in M \co d(p, q) < R_p$. \begin{proof}
		One side is trivial. For the other one, suppose there is some $\R\in R_p$ that bounds $d(p, q)$ for all $q \in M$. Fix some $p \in M$ (assume $M \neq \varnothing$, otherwise every number is trivially is bound), and let $q_1, q_2 \in M$. Then: 
		\[ d(q_1, q_2) \le d(q_1, p) + d(p, q_2) = d(p, q_1) + d(p, q_2) < 2R_p \]
		Hence $2R_p$ is a bound for the distance between each $q_1, q_2 \in M$. 
	\end{proof}
	
	\nsubsection{Napkin 6.1.6}
	In metric spaces, totally bound implies boundness. \begin{proof}
		Let $M$ be a totally bound space, meaning there are finitely many $\eg$-balls denoted as $U(x_i, \eg_i)$ that covers $M$. The union of balls is an open set, denotes as $U$, and each open set $U$ has a basic neighborhood (another ball) containing it. Finally there is an open ball $U(x, \eg)$ that contains the whole of $M$. In fact, for each $x, y \in M$ from triangle inequality $x, y \in U(x, \eg)$ implies $d(x, y) < 2\eg$. Finally $M$ is bounded. 
	\end{proof}

			\end{align*}
			We'll notice that for $A = \varnothing$ we get the discrete topology, and for $A = X$ we get the trivial topology. 
		\end{proof}

	\nsubsection{3H $\sim$ ${F_\sg}$ and ${G_\dg}$}
	A subset of a topological space $X$ is a $G_\dg$ iff it's a countable itersection of open sets, and an $F_\sg$ if it's a countable union of closed sets. 
	\begin{enumerate}
		\item The complement of a $G_\dg$ is an $F_\sg$ and vise versa. 
		\begin{proof}
			We'll proof one side, the other one is trivial from duality. Let $G \subset X$ be some $G_\dg$. We'll proof $\ol{G_\dg}$ is a $F_\sg$. Denote $(G_i)_{i = 1}^{\inft}$ be open sets such as $\cup G_i = A$ (exists be definition). From De-Morgan we get: 
			\[ \ol{G} = \ol{\bigcup G_i} = \bigcap \ol{G_i} =: \bigcap F_i \]
			Since $G_i$ is open, then $F_i := \ol{G_i}$ is closed. Hence $\ol{G} = \bigcap F_i$ is a finite intersection of closed sets, then $\ol{G}$ is a $F_\dg$ as interned. 
		\end{proof}
		\item An $F_\sg$ can be written as a union of countable $F_1 \subset F_2 \subset F_3 \subset \cdots$ of closed sets. 
		
		\item A closed set in a metric space if a $G_\dg$. \begin{proof}
			Suppose $A$ is some closed set in a metric space $(X, \rho)$. For the sake of simplicity I'll assume it's complement is an open disk (and not a union of open disk: but it's enough to look at the base of the topology). So $\ol A = U(x, \eg)$. Look at the closed set $U_n = [x, \frac{1}{n}]$: we notice that $\bigcap U_n = U(x, \eg)$. Hence $\ol A$ is an $F_\sg$, meaning $A$ is a $G_\dg$. 
			
			The general case can be deduce by taking apart $A$ to be a finite union of simple closed subsets (whom complement is an open disk) which is doable since the closed disks form a base for the family of closed sets in a metric space. 
		\end{proof}
		\item $\Q$ is $F_\sg$ in $\R$. \begin{proof}
			Since $\{r\}$ is closed for each $r \in \R$, and $\sof{\Q} = \az$, we won (because then we could jusk take the union of the singletons of each element $q \in \Q$). 
			
			Note that $\{r\}$ is closed because for each $a \in \R\setminus \{r\} = \ol{\{r\}}$, we get from density that exists such $b \in \R$ that (WOLOG) $r < b < a$, and then for $\dg = a - b$ we get that $(a - \dg, a + \dg)$ is an open disk (and in fact an open set) in $\R \setminus \{r\}$, so $\R\setminus \{r\}$ is open. 
		\end{proof}
	\end{enumerate}

	\begin{enumerate}
		\item the family of sets with the form $(-\inft, a)$ together with $(b, \inft)$ form a subbase for the usual topology. 
		\begin{proof}
			Denote with $\UU$ the family of all finite intersections of $(-\inft, a), (b, \inft)$. We'll proof it forms the usual base for the usual topology. Indeed, for each $(a, b)$ open disk from the base with have $(a, b) = (-\inft, b) \cap (a, \inft)$, and $\varnothing = (-\inft, 0) \cap (0, + \inft)$. On the other hand, for each $a, b$ the intersection of $(a, \inft)$ and $(\inft, b)$ is either $\varnothing$ if $a \le b$ or $(b, a)$ if $a > b$, both are in the usual topology, and the intersection of $(a, \inft)$ and $(b, \inft)$ is just $(\min(a, b), \inft)$ which is too open in the usual topology. Therefore each collection of finite elements from the subbase can be broken into pairs of two sets, whom their intersection is from the usual topology, that is closed for finite intersections, and consequently the intersection of that collection is in the usual topology.  
		\end{proof}
		\item Let the family of all straight lines in a plane to be a subbase. Then for each $r \in X$ we can take some two lines such that their intersection is $\{x\}$. Hence the family of all intersections is contained inside the set of singletons of $X$, which is the base for the discrete topology, the finest topology (so we don't care about any other intersections, and indeed that subbase is a subbase for exactly the discrete topology, and not some larger topology).

	\nsubsection{5F $\sim$ Second countable and separable spaces}
	A space $X$ is \textit{second countable} iff $X$ has a countable base. $X$ is separable iff a countable subset $D$ of $X$ exists with $\Cl_XD = X$ (such $D$ said to be \textit{dense} in $X$). 
	\begin{enumerate}
		\item A separable metric space is second countable. \begin{proof}
			Let some $(X, \rho)$ be a metric space with topology $\tau$. 
			Suppose $A$ is separable. So there is some countable $D$ with $\Cl_XD = X$. WOLOD $D \neq \varnothing$, since otherwise $\varnothing = \Cl_XD = X$ and each topology over $X$ is finite. Lemma: each open set meets $D$. Ad absurdum exists some $A$ open and $A \cap D = \varnothing$. Then by definition $\ol A$ is closed. Since $A \cap D = \varnothing$ and both $A, D \subset X$, we get $D \subset \ol A$. Finally $X = \Cl_X A \subset \ol A \subset X$, hence $A = X \supset D$, so $A$ meets $D$ at any of its point -- a contradiction. 
			
			Now, consider the one-to-one mapping $D\times\Q \ni (d, \eg) \mapsto U(d, \eg)$. Its image will be denoted with $\XX$. 
			
			Let $x \in X$. Clearly $\XX \subset \tau$ (it's a group of open disks). For each $U \in \tau$ open set, it's a union of $\bigcup \oc$ where $O \in \oc$ is a basic neighborhood (open disk) marked as $U(x_O, \eg_O)$. For each $\eg_O$ and $x \in U(x_O, \eg_O)$ and $d \in U(x_O, \eg_O)$ we have a rational $q^{O}_{d}$ having the property that $0 < q < \min\{\rho(x, d), \eg_O - \rho(x, d)\}$. 
			For that $O$ we'll define $\oc_O := \{U(d, q^{O}_{d}) \mid d \in D\}$. We immediately from definition and the triangle inequality have $U(d, q^{O}_d) \subset U(x_O, \eg_O)$. Moreover, for each $y \in U(x_O, \eg_O)$ and some open $P$ with $y \in U(y, \eg_O^{d}) \subset U(x_O, \eg_O)$ (same similar contraction), $P$ meets $D$ from the lemma at $d$. Both of them are in $U(y, \eg_O^{y})$ so $\rho(y, d) < \eg_O^{y} < \eg_O$. So we have some rational $q \in \Q$ that satisfies $0 <\eg_O^{y} < q < \min\{\rho(x, d), \eg_O - \rho(x, d)\}$, and finally we found a open disk $U(d, q) \in \XX$ with $y, d \in U(d, q) \subset U(y, \eg_O^{y}) \subset U(d, q_d^{O})$. Hence $\bigcup \oc_O = U(x_O, \eg_O)$. Then we have $\bigcup_{O \in \oc}\bigcup \oc_O = U$, and $\oc_O \subset \XX$, so $U$ is a union of items from (the now basis) $\XX$, since every open set in $\tau$ is given by a union (of union) of elements from $\XX$ and $\XX$ is contained in $\tau$. 
			
			We found a base $\XX$ that has one-to-one mapping into $D \times \Q$, which is countable because $D$ is countable and $\Q$ is countable, and $\az^{2} = \az$. Finally $\XX$ is a countable base for $(X, \rho)$, so $(X, \rho)$ is second-countable. 
		\end{proof}

			\begin{itemize}

As explained in the book, for each $A \in \UU_x$ we can take a open space $A\intr$ having the property $x \in A\intr \subset A$, and then $\UU_x' = \{A\intr \mid A \in \UU_x\}$ is a neighborhood system with open sets. Ergo $\UU_x' \subset \XX \mapsto \N$, so $\UU_x'$ is a countable neighborhood system. Finally we have a countable neighborhood system for each $x \in X$ as intended. 
				
				\textit{Note: I essentially needlessly repeated the proof shown from theorem 5.4. }
				\item \textbf{$\bm X$ is separable: }By AC there is a $F \co \XX \setminus \{\varnothing\} \to X$ choice function. We'll prove $D = \Img F$ is a countable dense set.

			\end{itemize}
		\end{proof}

	\nsubsection{7H $\sim$ Disjoint homeomorphism}
	Suppose some $(X, \tau_X), (Y, \tau_Y)$ topological spaces such that $X = \biguplus X_n, Y = \biguplus Y_n$ a disjoint union of open sets $X_n\in \tau_X, Y_n \in \tau_Y$, then if $X_n \overset{f_n}{\cong} Y_N$ are homeomorphic, then $X \cong Y$. \begin{proof}
		Consider a base for the relative topology at $X_n$, denoted $\XX_n$. Since $X_n, X_m$ are disjoint for each $n \neq m$, the union $\XX := \biguplus \XX_n$ is a base for $X$; for each open $U \in \tau$, we know that $U$ can be broken uniquely into $U =: \biguplus U_n$ such that $U_n = X_n \cap U$, and $U_n \in X_n$ is a union of elements from the base $\XX_n$. Finally the union of the union of all of these elements form $U$, hence $\XX$ indeed generated (via union) every open set in the topology $\tau$, and each set in $\XX$ is open (because the relative topology on $X_n$ is weaker than $\tau_X$). Similarly, we can build a base $\YY = \biguplus \YY_n$ for $Y$. 
		
		Since $f_n \co X_n \to Y_n$ is homeomorphism, it maps the open sets $\XX_n \mapsto \YY_n$. By defining: 
		\begin{align*}
			f \co X \to Y && f(x) = f_{\iota n \in \N \co x \in X_n}(x) = \biguplus f_n
		\end{align*}
		That is well-defined since $X_n, Y_n$ are a disjoint decomposition of $X, Y$, we have that $f$ maps $\XX \mapsto \YY$, as it's the union of $f_n$ that preserves $\XX_n \mapsto \YY_n$. Hence $f \co X \to Y$ is a homeomorphism, so $X \cong Y$. 
	\end{proof}
	
	\nsubsection{7L $\sim$ Linear operators and linear functionals}
	Let $(X, +_X, \cdot_X, \snorm_X),\ (Y, +_Y, \cdot_Y, \snorm_Y)$ over rings $\C$, be some normed (and in particular, metric and topological) linear spaces. 
	
	A \textit{linear operator $f \co X \to Y$} is a map that preserves $\cdot, +$. 
	
	An operator $\Cg$ is \textit{bounded} iff $\exists M \in \C \co \norm{\Cg(x)} \le M \norm x$. 
	
	\begin{enumerate}
		\item A linear operator is bounded iff $\sup \norm{S_1} < \inft$ where $\norm{S_1} := \{\norm{\Cg(x)} \co x \in X \land \norm x = 1\}$. \begin{proof}
			\begin{itemize}
				\item[$\impliedby$] Suppose the supremum of $\norm{S_1}$ is $\inft$ (note that $\norm{\Cg(x)} \ge 0$). Ad absurdum $\Cg$ is bounded by $M$. By supremum definition, exists a $x \in X$ where $\norm x > 1$ such that $\norm{\Cg(x)} > M$. But $\norm{\Cg(x)}$ is bounded by $M \cdot \norm{x} = M$, hence a contradiction. Thus, $\Cg$ must be bounded. 
				\item[$\implies$] Suppose $\Cg$ is bounded by some $M$. So for each $\norm x = 1$ we have $\norm{\Cg(x)} \le M \cdot \norm x = M$ for some $M$. Hence, $\norm{S_1}$ is bounded. By the completeness axiom, we know that a supremum for $S_1$ exists in $\C$, as intended. 
			\end{itemize}
		\end{proof}
		\textit{Note: }it can be proven that $\norm{S_1}$ is the supremum (or the maximum the the finite-dimensional case) of the singular values of $\Cg$. 
		\item TFAE for a linear $\Cg$: 
		\begin{enumerate}
			\item[a)]$\Cg$ is continues at some $x_0 \in X$
			\item[b)]$\Cg$ is uniformly continues on $X$
			\item[c)]$\Cg$ is bounded
		\end{enumerate}
		\begin{proof}
			\begin{itemize}
				\item[a$\to$b]Suppose $\Cg$ is continues at $x_0$. Note that continuity in the topological sense consolidates with continuity in the metric sense, since basic neighborhood sets in a metric space are open disks (and by using disks instead of neighborhoods, we get the same definition). 
				
				Then $\forall \eg > 0.\, \exists \dg > 0 \co \norm{\Cg(x_0 + \dg) - \Cg(x_0)} < \eg$. But we know $\Cg(x_0 + \dg) - \Cg(x_0) = \Cg(\dg) = \Cg(x_1 + \dg) - \Cg(x_1)$ by linearity, hence $\Cg$ is continues at all of $X$, for that very same $\dg$, hence uniformly continues. 
				\item[b$\to$c]Suppose $\Cg$ is uniformly continues on $X$. Then for all $x, y \in X$, for each $\eg \in \R$, we have that $\norm{x - y} < \dg$ for some $\dg$ implies that $\sof{\Cg(x) - \Cg(y)} < \eg$. Using linearity, fixing $y = 0_X$, and other nonsense we get: 
				\[ \norm{\Cg(x)} = \norm{\Cg(x -_X y)} = \norm{\Cg(x) -_Y \Cg(y)} < \eg \]
				In particular, for $\eg = 1$. Denote $v = \frac{\dg}{2 \norm x}x$. We know that $\norm{v} = \frac{\dg}{2\norm x} \norm x = \frac{\dg}{2} < \dg$. Therefore $\norm{\Cg(v)} < \eg = 1$. Now, we have from linearity: 
				\[ \frac{\dg}{2 \norm x}\norm{\Cg x} = \norm{\Cg\cl{\frac{\dg}{2 \norm x}x}} = \norm{\Cg v} < 1 \implies \norm{\Cg x} < \frac{2}{\dg} \cdot \norm x \]
				So $\frac{2}{\dg}$ is a constant bound for $x$, as intended. 
				\item[c$\to$a]Assume $\Cg$ is bounded. Prove it's continues at some $x_0 \in X$. 
					We'll prove for $x_0 = 0$. We know $\Cg(0) = 0$ as it's linear. Indeed, by boundness: 
					\[ \exists M \in \C.\, \forall x \in X \co  \norm{\Cg(x) - \Cg(0)} = \norm{\Cg(x)} \le M \norm{x} \]
					Hence, for each $\eg > 0$, we can fix an arbitrary small $\norm{y} < \frac{\dg}{M}$ (this can be done by exploiting some $\norm{y} \neq 0$, and inductively subtracting $\frac{y}{2^{n}}$), that will satisfy the condition for continuity at $0$ from the statement above. 
			\end{itemize}
		\end{proof}
	\end{enumerate}

				\item If $x = y$ then indeed $\ol{x} = \ol{y}$ so $x \sim x$. 
				\item If $x \sim y \land y \sim z$ then $\ol{x} = \ol{y} \land \ol{y} = \ol{z}$ therefore $x \sim z$. 
				\item If $x \sim y$ then $\ol{x} = \ol{y}$ thereupon we get $\ol{y} = \ol{x}$ so $y \sim x$. 
			\end{itemize}
			Finally $\sim$ is an equivalence relation. 
		\end{proof}
		\item Prove that $X/\!\!\sim\ =: \tl X$ is $T_0$. \begin{proof}
			Let $[x] \neq [y] \in X\eqrel$. Then either $x\notin \Cl \{y\} \lor y \notin \Cl\{x\}$, WOLOG assume the first. Because the closure is closed, $X \setminus \Cl \{y\}$ is an open set. It's obviously saturated since for the natural map $P \co X \to X\eqrel$ and $U =: X\eqrel \setminus \{[x]\}$, clearly $P\op(U) = X \setminus \Cl_X\{y\}$, hence $U$ is open. Clearly $[y] \in U \land [x] \notin U$, as intended. Finally $X\eqrel$ is $T_0$. 
		\end{proof}
	\end{itemize}

	\nsubsection{13G $\sim$ Topological groups}
	A \textit{topological group} is defined as a $T_2$ space such as the inversion $O$ and the multiplication $m$ maps are continues. The identity of the group is denoted as $e$. 
	\begin{enumerate}
		\item $m \co X\times X \to X$ is continues iff for each neighborhood $W$ of $xy$ there are neighborhoods $U$ of $x$ and $V$ of $y$ such that $UV \subset W$. \begin{proof}
			Trivial from the definition of continuity and product topology. 
		\end{proof}
		\item A group is a topological group iff $c \co G \times G \to G$ defined as $c(x, y) = x \cdot y\op$ is continues. \begin{proof}
			Clearly $O$ the inversion map is continues since $O = \pi_1\circ c$ is a composition of continues maps. 
			We also know that $m$ is continues, because the map $f(x, y) = (x, y\op)$ is continues because $\pi_1 \circ f = I_G$ which is continues and $\pi_2 \circ f = O$ which is continues too. Finally we get that $m = c \circ f$, and a composition of continues map is continues. 
		\end{proof}
		\item \begin{itemize}
			\item In $\R$, which regular addition and usual topology, we get a topological group. \begin{proof}
				First of all $O(x) = -x$ is continues because $(a, b) \mapsfrom (-a, -b)$ both are open (in fact, $O$ is an homeomorphism). Moreover $m(x, y) = x + y$ is continues, because $(x_n, y_n) \to (x, y)$ implies $f(x_n, y_n) = x_n - y_n \to x - y = f(x, y)$ and $\R$ is first countable. Finally addition and inversion are continues as intended. 
			\end{proof}
			\item Each group with the discrete topology is a topological group. \begin{proof}
				Consider some group $G$ with the discrete topology and some $O \co G \to G$ inversion and $m \co G \times G \to G$ multiplication map. $O$ is continues because each operator within the discrete topology is continues. $G \times G$ has the discrete topology too, therefore $m$ is also continues. Finally $G$ is a topological group. 
			\end{proof}
			\item Let $a, b \in G$ a topological group. Each of the maps: 
			\begin{align*}
				(1) \ x \to x\op && (2) \ x \to ax && (3) \ x \to xb && (4) x \to axb
			\end{align*}
			It's sufficient to prove that $(1)$ and $(2)$ are homeomorphisms because $(4)$ is composition of $(2)$ and $(3)$ and $(3)$ can be proven from $(2)$ in the group that has $m'(x, y) = m(x, y)$. 
			
			For $(1)$, indeed $O(x) = x\op$ is continues. It's well known that inverse is unique in group theory (because let some $a, b\in G$, we get that if $O(a) = O(b)$ thereupon $O(O(a)) = O(O(b))$ and inversion is assumed to be monoidical so $a = b$, and more generally naturally isomorphic to it). Its inverse is also continues, because $O \circ O = I$ therefore $O = O\op$, hence both $O\op$ is open (it's given that $O$ is open). Finally $O$ is an homeomorphism. 
			
			For $(2)$, fix some $c \in G$ and define $f(x) = c \cdot x$. It's a bijection because if $f(x) = f(y)$ we know $cx = cy$, and by multiplying from the right by $c\op$ we get $c\op c x = c\op c  y$ so $x = y$. Also for each $x \in G$ we gave $f\op(x) = c\op x$ because $f(c\op x) = cc\op x = x$. Indeed its injective and surjective. 
			
			Clearly $m|_{\{c\} \times G}$ is continues and a restriction of a continues map is continues, and $\{c\} \times G \overset{r}{\cong} G$, finally $f = r \circ m|_{\{c\} \times G}$. Same goes for $c\op = c\op x$, $\{c\op\}\times G \overset{l}{\cong} G$ and $f\op = l \circ m|_{\{c\op\} \times G}$. 
			
			Finally $f$ is bijection, continues and its inverse is continues too. 
		\end{itemize}
		\item If $\UU$ is a neighborhood base at $e$, then for all $x \in G$, also $\{xU \mid U \in \UU\}$ and $\{Ux \mid U \in \UU\}$ are neighborhood bases at $x$. \begin{proof}
			Let $x \in G$, and let some $\UU$ neighborhood base for $x$. We'll prove $x\UU$ is a neighborhood base at $x$. Consider the map $f(y) = xy$. Clearly $f(e) = x$ and $f(\UU) = x\UU$. Because we proved $f$ is a homeomorphism, for each neighborhood $O$ at $x$ we know that $f\op(O)$ is neighborhood at $e$ hence we have some $\UU \ni U \subset O$. Then $x\UU \ni xU = f(U) \subset f(f\op(O)) = O$ so indeed by definition $x\UU$ is a basic neighborhood. Similarly for $\UU x$. 
		\end{proof}
	\end{enumerate}
