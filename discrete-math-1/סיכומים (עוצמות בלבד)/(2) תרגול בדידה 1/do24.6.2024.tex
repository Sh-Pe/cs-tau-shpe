	\section{}
	\textbf{הוכח/הפרך: }קיימת $X \subseteq \ps(\N)$ כך ש־$|X|= \az$ כך שלכל $A, B \in X. A \cap B = \emptyset$ וגם $\N = \bigcup_{A \in X}A$
	
	\begin{proof}
		נסמן $\P$ כקבוצת הראשוניים, ונבחר: 
		\[ A_p = \{p^n \mid n \in \N_+\}, \ X = A_p \mid p \in \P\} \cup \{\N \setminus \bigcup_{p \in \P}A_p\} \]
		\begin{enumerate}
			\item נוכיח $|X| = \az$. החח"ע הבא:
			\[ f \colon \P \to X, \ f = \lambda p \in \P. A_p \implies \az \le |X| \]
			כיוון שני מהכלה. 
			\item זרות בזוגות: 
			הפרדה למקרים + מקרה שני, נניח בשלילה $n \in A_P \cap A_q$ולכן $n = p^{k} = p^{\ell}$ וזו סתירה למשפט הסידוי של האריתמטיקה
			\item איחוד טרוויאלי
			\item $\forall A \in X. |X| = \az$: $n \mapsto p^{n}$ חח"ע ועל, וגם $B = \N \setminus \bigcup_{p}A_p$ויתקיים $|B| \le \az$ מכלה. 
		\end{enumerate}
	\end{proof}
	\section{}
	מצאו את עוצמת הקבוצה: 
	\[ A = \{f \colon \N \to \N \mid \forall n, m \in \N. n < m \implies f(n) > f(m)\} \]
	\textbf{טענה: }אם $f \in A$, יש $n_0 \in \N$ ו־$c \in \N$ כך ש־$\forall n \ge n_0. f(n) = c$. 
	
	נוכיח $|A| \le \az$. נתבונן ב־
	$A_{n, c} = \{f \in A \mid \forall k \ge n. f(k) = c\}$
	טענה: 
	\[ A = \bigcup_{n, c \in \N}A_{n, c} \]
	
	נרצה להוכיח $|A_{n, c}| \le \az$. נראה פונקציה חח"ע $A_{n, c} \to \N^{n}$: 
	\[ f = \lambda g \in A_{n, c}. \la g(0), \dots, g(n - 1) \ra \]
	
	וזו קבוצת בת־מנייה ($|\N^{n}| = \az$). מאיחוד לפחות בן־מניה של קבוצות לפחות בנות מניה נקבל את הדרוש.
