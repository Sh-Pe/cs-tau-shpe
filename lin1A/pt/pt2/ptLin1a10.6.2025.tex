    נוכיח ש־$\ker \phi_3 \subseteq \Img T_2$ בלמה של החמישה. 
    
    \begin{proof}
        יהי $v_3 \in \ker \phi_3$. נסמן $v_u = T_3v_3$ וכן $u_4 = \phi_4(v_u)$. מקומטטיביות: 
        \[ u_4 = \phi_4(v_u) = \phi_u(T_3(v_3)) = S_3 (\phi_3(v_3)) = S_3(0) = 0 \implies v_4 = 0 \implies T_3v_3 = 0 \implies v_3 \in \ker T_3 = \Img T_2 \]
    \end{proof}
    
    נוכיח $\ker\phi_3 = \{0\}$. 
    
    \begin{proof}
        עתה נראה ש־$\ker \phi_3 = 0$. יהי $v_2 \in V_2$ כך ש־$T_2v_2 = v_3$, נסמן $u_2 = \phi_2v_2$ אז מתקיים: 
        \[ S_2u_2 = S_2\phi_2 v_2 = \phi_3 T_2v_2 = \phi_3v_3 = 0 \]
        לכן $u_2 \in \Img S_1$. כלומר ישנו $u_1 \in U_1$ כך ש־$S_1u_1 = u_2$. נלך אחורה ע''י הגדרת $v_1 \ \phi_1\op u_1$. כמו כן
        \[ \phi_2 v_2 = u_2 = S_1u_1 = S_1\phi_1v_1 = \phi_2T_1v_1 \implies v_2 = T_1v-1 \implies v_2 \in \Img T_1 = \ker T_2 \implies T_2v_2 = 0 \]
    \end{proof}
    
    נוכיח  $\forall u_3 \in U_3 \co \exists v_3 \in V_3 \co (\phi_4 \circ T_3)v_3 = S_3u_3$.
    \begin{proof}
        באופן שקול $\phi_4\op(S_3(u_3)) = \ker T_4$ שקול לטענה $\forall u_3 \in U_3 \co \exists v_3 \in V_3 \co (\phi_4 \circ T_3)v_3 = S_3u_3$. ממדויקות שקול ל־$(\phi_4\op \circ S_3)u_3 \in \ker T_4$ כלומר $0 = (T_4 \circ \phi\op_4 \circ S_3)u_3$. שוב ממדויקות נשים לב ש־$0 = (S_4 \circ S_3)u_3$. מאחר ש־$\phi_5$ אזי נוכל לעלות למעלה $(\phi_5\op \circ S_4 \circ S_3)u_3 = \phi\op_5(0) = 0$. לבסוף מקומטטביות הריבוע האחרון: 
        \[ (T_4 \circ \phi_4\op = \phi\op_5 \circ S_4 \implies (T_4 \circ \phi\op_4 \circ S_3))u_3 = (\phi_5\op \circ S_4 \circ S_3)u_3 \]
    \end{proof}
    
    נותר להוכיח ש־$\phi_3$ איזו'. 
    \begin{proof}
        נרד למטה מהגדרת $\tl u_3 = \phi_3 v_3$. מקומטטיביות $S_3u_3 = S_3\tl u_3$. ממדויקות $S_3(u_3 - \tl u_3) = 0$ ולכן ממדוייקות $u_3 - \tl u_3 \in \Img S_2$. אזי ישנו $u_2 \in U_2$ כך ש־$S_3u_2 = u_3 - \tl u_3$. נלך למעלה ע''י הגדרת $v_2 = \phi_2\op u_2$. מקומטטיביות מתקיים 
        
        \[ \phi_3 T_2 v_2 = S_2 \phi)2 v_2 = S_2u_2 = u_3 - \tl u_3 = ו_3 - \phi_3v_3 \]
        נעביר אגפים ונקבל: 
        \[ u_3 = \phi_3(T_2v_2 + v_3) \]
        וסה''כ מצאנו מקור ל־$u_3$. כלומר $\phi_3$ על. 
    \end{proof}
    
    סה''כ היא איזו'. והוכחנו את עלמה כדרוש.

    \subsubsection*{בלי קשר}
    כמו איזומורפיזמים ישנם $T \co \F_q^{4}\to \F_1^4$? מספיק להגדיר איזו' על הבסיס הסטנדרטי: 
    ,\[ T(e_1) = v_1, \ \dots \ , T_(e_4) = v_4 \]
    נספור כמה אפשרויות כל וקטור נותן: 
    \begin{itemize}
        \item עבור $v_1$ נוכל לבחור כל וקטור לא אפס – $q^{4} - 1$. 
        \item עבור $v_2$ נוכל לבחור כל וקטור שאינו ת''ל ב־$v_1$. יש $q^4 - q$ אפשרויות. 
        \item עבור $v_3$ יהיו באופן דומה $q^4 - q^2$
        \item עבור $v_4$ יהיו $q^4 - q^3$
    \end{itemize}
    סה''כ: 
    \[ \prod_{i = 0}^{n - 1}q^n - q^i = q^{n(n - 1)} - \frac{q^{n} - 1}{q - 1} \]
