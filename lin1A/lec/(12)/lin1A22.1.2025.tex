	\textbf{טענה. }תהי $A \in M_n(\F)$. נסמנה ב־$(a_{ij})$. $\forall 1 \le i \le n \co |A| = \sum(-1)^{i + j}a_{ij}|A_{ij}|$
	כאשר $A_{ij}$ הוא המינור של $A$.
	
	\textit{הוכחה. }נסמן ב־$A'$ את המטריצה $A$ לאחר החלפת שורות $i, n$. אז $|A| = -|A'|$. עתה, ידועה הנוסחה: 
	\[ |A| = -|A'| = -\sum_{j = 1}^{n}(-1)^{n + j}a_{ij}\sof{A_{ij}} \]
	נשים לב ש־$a'_{nj} = a_{ij}$ וגם $A'{nj}$ הוא כמו $A_{ij}$ עם $|n - 1| - i$ החלפות. אז $\sof{A_{ij}} = (-1)^{(n - 1) - i}\sof{A'_{ni}}$. נציב ונקבל $|A| = \sum_{j = 1}^{n}(-1)^{n - 1- i}|A_{ij}$ כרצוי. 
	
	באופן דומה אפשר לעשות פיתוח לפי שורות: 
	\[ |A| = |A^T| = \text{לפי שורה} = \text{כרצוי} \]
	\section{\en{Block Matrix}}
	מטריצת בלוקים: $\pms{A & B \\ 0 & C}$. 
	\textbf{טענה. }תהינה $A \in M_n(\F), B \in M_{m \times n}(\F), D \in M_{m}(\F)$, אז $\det\pms{A & B \\ 0 & D} = \det A \det D$
	\begin{proof}
		נסמן ב־$\phi_1 \cdots \phi_t$ פעולות דירוג של מטריצה $A$, כלומר $|A| = |A'| \cdot \prod E_i$  עבור $E_i$ מטריצות אלמנטריות מתאימות לדירוג, ו־$A'$ מדורגת קאנונית שקולה ל־$A$. קיים ממשפט לפיו פעולות דירוג שקולות להכפלה במטריצה אלמנטרית, ושדיטרמיננטת מכפלת מטריצות היא מכפלת הדיט'. 
		
		נסמן $\phi_{t + 1} \cdots \phi_s$ פעולות דירוג כמקודם, אז קיימות: 
		$|D| = D' \prod E_i$ באופן דומה. נסמן $\phi'_1 \cdots \phi'_t$ פעולות המתאימות ל־$\phi_1 \cdots \phi_s$ בעבור מטריצה מגודל $(m + n) \times (m + n)$, כלומר, כל פעולה תפעל רק על $n$ השורות הראשונות, אבל על שורות בגודל $m + n$\. בדומה, נסמן ב־$\phi'_{t + 1} \cdots \phi'_s$. אז, כאשר $E'_1 \cdots E'_{s}$ מתאימות ל־$\phi'_1 \cdots \phi'_s$. 
		\[ \csb{\pms{A & B \\ 0 & D}} = \prod |E'_i| \underbrace{\csb{\pms{A' & * \\ 0 & D'}}}_{GG} \]
		
		אם $A$ או $D$ לא הפיכה, אז $|X|$ עבור $X$ המטריצה הלא הפיכה היא $0$, כלומר $X' \neq I$ קאנוני, ולכן $X'$ עם שורת אפסים, וסה"כ $GG$ עם שורת אפסים וסה"כ הדיט' שווה ל־0. 
		
		אחרת, $A, B$ הופכיות. נקבל $A' = I_n, D'=I_m$ וקיבלנו $\det GG = 1$ ובתור מט' משולשית עליונה, נקבל סה"כ שוויון ל־$\csb{\pms{A & B \\ 0 & D}}\prod |E_i| = |A| |D|$. 
		
		\textbf{הערה של קרני: }לא באמת צריך את ההפרדה למקרים. 
	\end{proof}
	
	\section{\en{Adjugate matrix}}
	\textbf{הגדרה. }תהי $A \in M_n(\F)$ מטריצה שתסומן ב־$(a_{ij})$. נגדיר את \textit{המטריצה המצורפת} (לעיתים נקראת מוצמדת) להיות המטריצה כך ש־: 
	\[ \adj(A)_{ij} = (-1)^{i + j}|A_{ji}| \]
	\textbf{דוגמה. }
	\[ \adj\pms{a  & b \\ c & d} = \pms{(-1)^{2}d & (-1)^{2 + 1}b \\ \cdots & \cdots} = \pms{d & -b \\ -c & a} \]
	\textbf{משפט. }תהי מטריצה $A \in M_n(\F)$. אז $A \cdot \adj A = \adj A \cdot A = |A| \cdot I$. ובפרט, עבור $A$ הפיכה, מצאנו נוסחא להופכית. כלומר $A\op = \frac{1}{|A|} \cdot \adj A$. 
	\begin{proof}
		\textbf{למה. }יהיו $1 \le i \le j \le n$. אז $\sum^n_{k = 1} (-1)^{j + k} a_{ik}|A_{jk}| = |A|\delta(i, j)$
		כאשר $\delta$ היא הדלתא של קרונקר (פונ' $\chi$). 
		\begin{proof}[תת־הוכחה]
			אם $i \neq j$: נתבונן ב־$A'$, המטריצה $A$ שהשורה ה־$j$ שלה שווה לשורה ה־$i$. ברור כי $\det A' = 0$. נפתח לפי השורה ה־$j$ של $A$. נבחין בכך ש־$A'_{jk} = A_{jk}$, וגם $a'_{jk} = a_{ik}$. 
		\[ 0 = |A'| = \sum (-1)^{j + k} a'_{jk}|A'_{jk}| = \sum (-1)^{j + k}a_{ik}|A_{jk}| \]
		
		אחרת $i = j$ ואז "ברור" (אוטומטית פיתוח). 
		\end{proof}
		
		נחזור למשפט. 
		\[ (A \adj A)_{ij} = \sum_{k = 1}^{n}a_{ik}(\adj A)_{kj} = \sum a_{ik}(-1)^{k + j}|A_{jk}| = |A|\delta(i, j) = (|A|I)_{ij} \]
		וואפי וואפי	
		
		עתה נראה ש־$\adj A \cdot A = |A| I$. נסמן $B = (A^T)$. אז: 
		\[ \adj B \cdot B = |B| \cdot I \implies \underbrace{A^T \adj A}_{\mathclap{(\adj A^T)^T \cdot (A^T)^T = (I|A|)^T = |A| I}} = \overbrace{|A^T|}^{\mathclap{=|A|}} \cdot I \]
	\end{proof}
