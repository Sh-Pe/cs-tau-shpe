	\section{\en{Cum Sum}}
	\sen (they knew) \she
	
	שאלה מתוך המבחן. תנו דעתכם על העץ הבא: 
	\begin{center}
		\sen\begin{forest}
			[5[2[3]][9[8[6]][12]]]
		\end{forest}
		$\implies$
		\begin{forest}
[10[2[5]][33[24[16]][45]]]
		\end{forest}
		\she
	\end{center}
	תכונה: כל המפתחות בתת עץ הימני, עבור צומת נתון, גדולים ממנו, וכל המפתחות בתת העץ השמאלי של צומת נתון, קטנים ממנו. התכונה הזו נכונה לכל עץ חיפוש בינארי. 
	
	נרצה להגדיר sumsum כפונקציה שמשנה את ערך המפתח של כל צומת להיות סכום כל המפתחות שתחתיו. דוגמה להפעלת הפעולה מצויירת לעיל.
	
	שאלה כללית: איך למיין עצי חיפוש?
	\begin{itemize}
		\item in-order: שמאל $\leftarrow$ אני $\leftarrow$ ימין
		\item pre-order: אני $\leftarrow$ שמאל $\leftarrow$ ימין
		\item post-order: שמאל $\leftarrow$ ימין $\leftarrow$ אני
	\end{itemize}
	את כולם אפשר לממש בזמן ריצה ליניארי. ב־cumsum נרצה לעשות מיון in-order. ניעזר בתבנית שניתנה במבחן: 
	
\sen
\begin{lstlisting}
def cumsum(self): 
	def cumsum_rec(node, s) -> int: 
		# edge case
		if node is None: 
			return s
		s = cumsum_rec(node.left, s)
		node.key += s
		s = cumsum_rec(node.left, s)
		node.key += s
		s = cumsum_rec(node.right, node.key)
		return s
	
	cumsum_rec(self.root, 0)
\end{lstlisting}
\she
איך מגיעים לפתרון: כאשר מטיילים על עץ חיפוש בינארי, יש לנו סדר שאפשר לעבוד איתו לפתור ליניארית. 

	שאלה: איך היה אפשר לסדר את העץ הזה מחדש, כך ש־pre-order היה מחזיר את הכל ממוין: תשובה: סרוך שהולך ימינה. 
	
	\section{\en{Generators}}
	שאלות על גנרטורים מהמבחן.
	\textbf{הגדרה: }גנרטור הוא בעל השהייה סופית אמ"מ כל קריאה ל־next מסתיימת תוך זמן סופי. בכל סעיף, צריך להגיד האם ניתן לממש אותו בעזרת גנרטור עם השהייה סופית 
	\begin{enumerate}
		\item ייצור מספרים ראשוניים – \textbf{אפשר}. הגנרטור זוכר מהראשוני האחרון שהוא בדק נעבור על כל הטבעיים, ונחזיר כאשר נמצא ראשוני. בין שני ראשוניים יש מספר סופי של מספרים. 
		\item בהינתן $g$ עם השהייה סופית שמחזיר מספרים, להחזיר את איברי $g$ החיוביים. \textbf{אי־אפשר}. דוגמה נידית: נתבונן ב־$g = \lambda n \in \N. -1$, ואז הגנרטור יאלץ לעשות $\inf$ פעולות על מציאת איבר חיובי (כי אין כאלו). 
		\item בהינתן $g$ עם השהייה סופית שמחזיר מספרים, להחזיר את הסכומים החלקיים של $g$. \textbf{אפשר}. אפשר לשמור את הסכומים החלקיים שהוחזרו קודם וכל פעם מוסיפים את מה שהיה קודם. 
	\end{enumerate}
